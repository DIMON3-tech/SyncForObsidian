Левый бинпоиск ищет первое вхождение числа
Пример найти первое вхождение х == 6
```cpp
#include <iostream>  
  
int main() {  
    int arr[6] = {1, 2, 3, 6, 6, 7};  
    int l = 0;  
    int r = 5;  
    int x = 6;  
    while (l + 1 < r) {  
        int middle = l + (r - l) / 2;  
        //std::cout << middle << std::endl;  
        if (x < arr[middle]) {  
            r = middle;  
        } else  if(x > arr[middle]){ // даже если равно, все равно сдвигаться вправо, чтобы найти последнее  
            l = middle;  
        } else {  
            r = middle;  
        }  
    }  
    if (arr[r] == x) {  
        std::cout << r << std::endl;  
    }  
    if (arr[l] == x) {  
        std::cout << l << std::endl;  
    }  
  
}
```
