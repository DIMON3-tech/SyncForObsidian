void pushToEnd(int _value){  
    Node* p = new Node(_value);  
    if(is_empty()){  
        head = p; // следующий элемент этот же самый (больше нет элементов)  
        tail = p; // последний элемент этот же самый (больше нет элементов)  
        return;  
    }  
    tail->next = p;  
    tail = p;  
    tail->next = nullptr; // не уверен, но работало и без него  
  
}
